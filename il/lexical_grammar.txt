expression : literal | grouping | unary | binary ;

literal    : NUMBER | STRING | "true" | "false" | "null" ;

grouping   : "(" expression ")" ;

unary      : ("-" | "!") expression ;

binary     : expression operator expression ;

operator   : "-" | "+" | "/" | "*" | "!=" | ">" | ">=" | "<" | "<=" | "==" | "=" ;
